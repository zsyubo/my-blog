# 指针汇编代码分析

源代码

```c++
	int age = 3;
	int* p = &age;
	*p = 5;
```

汇编代码

```assembly
// int age = 3;
010B27BB  mov         dword ptr [age],3  
// 把 age的地址给 eax，  lea指令是吧地址值赋值给eax
010B27C2  lea         eax,[age]  
// 把地址值给  p， 所以p存的是地址值
010B27C5  mov         dword ptr [p],eax  
// 取p的数据(age的地址)给eax。
010B27C8  mov         eax,dword ptr [p]  
// 写入到 age。
010B27CB  mov         dword ptr [eax],5  
```
